roles(role_id PK, role_name UNIQUE NOT NULL)

users(user_id PK, nickname UNIQUE NOT NULL, email UNIQUE NOT NULL, role FK→roles.role_id)

moderator_info(user_id FK→users.user_id, seniority_level CHECK 1–3, moderation_area)

psychologist_info(user_id FK→users.user_id, license_number UNIQUE NOT NULL, specialization)

interaction_types(type_id PK, type_name UNIQUE NOT NULL)

interaction(interaction_id PK, user_id FK→users.user_id, interaction_type FK→interaction_types.type_id, interaction_date, platform, text NOT NULL, isAbusive)

case_state(state_id PK, state_name UNIQUE NOT NULL)

cyberbullying_case(case_id PK, interaction_id FK→interaction.interaction_id, moderator_id FK→users.user_id, severity CHECK 1–5, region, open_date, close_date, comments, state FK→case_state.state_id)

intervention(intervention_id PK, case_id FK→cyberbullying_case.case_id, psychologist_id FK→users.user_id, intervention_type, intervention_date, notes)

resources(resource_id PK, title NOT NULL, url NOT NULL, format)

resource_access_log(access_id PK, resource_id FK→resources.resource_id, access_date)
